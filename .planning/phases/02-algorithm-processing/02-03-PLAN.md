Create src/stores/algorithm.ts:
```typescript
import { writable } from 'svelte/store';

interface AlgorithmState {
  steps: AlgorithmStep[];
  confirmed: boolean;
  isProcessing: boolean;
}

function createAlgorithmStore() {
  const { subscribe, set, update } = writable<AlgorithmState>({
    steps: [],
    confirmed: false,
    isProcessing: false
  });

  return {
    subscribe,
    setSteps: (steps: AlgorithmStep[]) => update(s => ({ ...s, steps, confirmed: false })),
    confirm: () => update(s => ({ ...s, confirmed: true })),
    reset: () => set({ steps: [], confirmed: false, isProcessing: false }),
    setProcessing: (isProcessing: boolean) => update(s => ({ ...s, isProcessing }))
  };
}

export const algorithmStore = createAlgorithmStore();
```
